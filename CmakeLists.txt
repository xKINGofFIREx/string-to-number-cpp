cmake_minimum_required(VERSION 3.5)
project(StringToNumber VERSION 0.1)
add_executable(StringToNumber WIN32 ${CMAKE_CURRENT_SOURCE_DIR}/src/main.cpp)

target_sources(StringToNumber PRIVATE ${CMAKE_CURRENT_SOURCE_DIR}/src)
target_include_directories(StringToNumber
                           PUBLIC ${CMAKE_CURRENT_SOURCE_DIR}/include)
